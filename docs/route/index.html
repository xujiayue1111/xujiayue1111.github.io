vue-router配置

下载路由插件

pnpm install vue-router

在src目录下创建views文件夹存放不同的路由应该显示的页面

在src目录下创建router文件夹存放对路由的配置，其中的index.ts配置路由模板，routes.ts存放常量路由。

index.ts​

// 通过vue-router插件实现模板配置
import { createRouter, createWebHashHistory } from 'vue-router'
import { constantRoute } from './routes'
// 创建路由器 在createRouter路由里面写一下配置项
let router = createRouter({
    // 路由模式hash 需要额外引入
    history:createWebHashHistory(),
    routes:constantRoute,
    // 添加滚动行为
    scrollBehavior(){
        return {
            left:0,
            top:0
        }
    }
})
export default router

routes.ts​

// 对外暴露常量路由
export const constantRoute = [
    // 登录路由
    {
        path:'/login',
        component:()=>import('@/views/login/index.vue'),
        name:'login' //给每一个路由添加命名空间 命名路由
    },
    // 根目录路由
    {
        path:'/', //就是根目录
        component:()=>import('@/views/home/index.vue'),
        name:'layout'
    },
    {
        // 404
        path:'/404',
        component:()=>import('@/views/404/index.vue'),
        name:'404'
    },
    {
        // 任意路由 上面的三个都没有匹配上就会访问这个任意的路由
        path:'/:pathMatch(.*)',
        redirect:'/404',
        name:'Any'
    }
]

在入口文件注册路由​

// 引入路由进行注册 
import router from './router'
// 注册路由
app.use(router)
